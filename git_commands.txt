
1. Работа с локальным репозиторием:

	git status # проверка статуса файлов в репозитории
		untracked - не отслеживается
		tracked - отслеживается
		staged - готов к коммиту
		modified - изменен

	git add 'имя файла' # добавляет файл в stages (tracked)
	git add --all # добавляет все файлы

	git commit -m 'Запись о коммите' # фиксирует новое состояние файла с меткой объявленного коммита
